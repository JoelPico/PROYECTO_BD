DECLARE 
    CURSOR NUMCLIENTES IS 
    SELECT * FROM CLIENTE;
    
    --cedu_cliente NUMCLIENTES%ROWTYPE;
    num_cliente NUMCLIENTES%ROWTYPE;
    --ape_cliente NUMCLIENTES%ROWTYPE;
    
    
BEGIN
    
    OPEN NUMCLIENTES;
    LOOP
        FETCH NUMCLIENTES INTO num_cliente;--,ape_cliente;
        EXIT WHEN NUMCLIENTES%NOTFOUND;
        DBMS_OUTPUT.PUT_LINE('-CEDULA: '||num_cliente.CI_CLIENTE||' -NOMBRE: '||num_cliente.NOMBRE_CLIENTE||' -APELLIDO: '|| num_cliente.APELLIDO_CLIENTE);--||'CANTIDAD'||ape_cliente.APELLIDO_CLIENTE);
    END LOOP;
    
    CLOSE NUMCLIENTES;
END;

--ACTUALIZA LOS DATOS ESPECIFICOS DE UNA TABLA


DECLARE 
v_numEmple NUMBER :=12;
v_nom VARCHAR(40) := 'ANA';
numero_empleado_invalido EXCEPTION;
BEGIN
UPDATE VENDEDOR
SET NOMBRE_VENDEDOR = v_nom
WHERE id_vendedor = v_numEmple;
IF SQL%NOTFOUND THEN
RAISE numero_empleado_invalido;
END IF;
COMMIT;
EXCEPTION
WHEN numero_empleado_invalido THEN
DBMS_OUTPUT.PUT_LINE('No existe el vendedor');
END;

--INDICA LA CANTIDAD DE VENDEDORES QUE TENEMOS ACTUALMENTE
DECLARE 
    num_vendedor NUMBER(12);
    muchos EXCEPTION;
    nohay EXCEPTION;
    
BEGIN
    SELECT count(1) INTO num_vendedor
    FROM VENDEDOR;
    
    IF num_vendedor > 10 THEN
        RAISE muchos;
    END IF;
    
    IF num_vendedor < 1 THEN
        RAISE nohay;
    END IF;
    
    DBMS_OUTPUT.PUT_LINE('HAY '||num_vendedor||' VENDEDORES');
    
EXCEPTION
    WHEN muchos THEN
    DBMS_OUTPUT.PUT_LINE('MUCHOS REGISTROS');
    WHEN nohay THEN
    DBMS_OUTPUT.PUT_LINE('NO HAY REGISTROS');
END;
